\noindent
{\large \textsf{\underline{Osnovne trigonometrijske formule:}}}\\
\hspace*{30mm}	\fbox{$\displaystyle \textrm{sin}^2\alpha+\textrm{cos}^2\alpha =1$}
		\hspace{5mm}
		\fbox{$\displaystyle \textrm{tg} \alpha=\frac{\textrm{sin} \alpha}{\textrm{cos} \alpha}$}
		\hspace{5mm}
		\fbox{$\displaystyle \textrm{ctg} \alpha=\frac{\textrm{cos} \alpha}{\textrm{sin} \alpha}$}

		\hspace{5mm}
		\fbox{$\textrm{cos}(-\alpha)=\textrm{cos}\alpha$}
		\hspace{5mm}
		\fbox{$\textrm{tg}(-\alpha)=-\textrm{tg}\alpha$}
		\hspace{5mm}
		\fbox{$\textrm{ctg}(-\alpha)=-\textrm{ctg}\alpha$}

	\item \textbf{\textsf{Izra\v{z}avanje jedne trigonometrijske funkcije pomo\'{c}u druge:}}\\
	\hspace{2cm}\fbox{$\displaystyle
	\textrm{sin}\alpha=\sqrt{1-\textrm{cos}^2\alpha}=\frac{\textrm{tg}\alpha}{\sqrt{1+\textrm{tg}^2\alpha}}=\frac
	1{\sqrt{1+\textrm{ctg}^2\alpha}}$} \hspace{3mm}\fbox{$\displaystyle \cos\alpha=\sqrt{1-\textrm{sin}^2\alpha}=\frac{1}{\sqrt{1+\textrm{tg}^2\alpha}}=\frac {\textrm{ctg}\alpha}{\sqrt{1+\textrm{ctg}^2\alpha}}$}\\
	\fbox{$\displaystyle \textrm{tg}\alpha=\frac{\sin\alpha}{\sqrt{1-\sin ^2\alpha}}=\frac{\sqrt{1-\cos ^2 \alpha}}{\cos \alpha}=\frac1{\textrm{ctg}\alpha}$}\hspace{4mm}
	\fbox{$\displaystyle \textrm{ctg}\alpha=\frac{\sqrt{1-\sin ^2\alpha}}{\sin\alpha}=\frac{\cos \alpha}{\sqrt{1-\cos ^2 \alpha}}=\frac1{\textrm{tg}\alpha}$}\\
	\item \textbf{\textsf{Funkcije zbroja i razlike kutova:}}\\
		\fbox{$\sin (\alpha \pm \beta)=\sin \alpha \cos \beta \pm \cos \alpha \sin \beta$}
		\hspace{1cm}
		\fbox{$\cos (\alpha \pm \beta)=\cos \alpha \cos \beta \mp \sin \alpha \sin \beta$}\\
		\hspace*{10mm}\fbox{$\displaystyle \textrm{tg}(\alpha \pm \beta)=\frac{\textrm{tg}\alpha \pm \textrm{tg}\beta}{1\mp \textrm{tg}\alpha \cdot \textrm{tg}\beta}$}
		\hspace{10mm}\fbox{$\displaystyle \textrm{ctg}(\alpha \pm \beta)=\frac{\textrm{ctg}\alpha \cdot \textrm{ctg}\beta \mp 1}{ \textrm{ctg}\beta \pm \textrm{ctg}\alpha}$}
	\item \textbf{\textsf{Funkcije dvostrukih kutova:}}
		\fbox{$\sin 2\alpha=2\sin \alpha\cdot \cos \alpha$} 
		\hspace{2mm}
		\fbox{$\cos 2\alpha=\cos^2\alpha - \sin ^2 \alpha$}\\
		\hspace*{52mm}\fbox{$\displaystyle \textrm{tg}2\alpha=\frac{2\textrm{tg}\alpha}{1-\textrm{tg}^2\alpha}$}
		\hspace{2mm}
		\fbox{$\displaystyle \textrm{ctg}2\alpha=\frac{\textrm{ctg}^2\alpha-1}{2\textrm{ctg}\alpha}$}
	\item \textbf{\textsf{Funkcije polovi\v{c}nog kuta:}}
		\fbox{$\displaystyle \sin \frac \alpha 2=\sqrt{\frac{1-\cos \alpha}2}$}
		\hspace{2mm}
		\fbox{$\displaystyle \cos \frac \alpha 2=\sqrt{\frac{1+\cos \alpha}2}$}\\
		\fbox{$\displaystyle \textrm{tg}\frac\alpha 2=\sqrt{\frac{1-\cos \alpha}{1+\cos\alpha}}=\frac{1-\cos \alpha}{\sin\alpha}=\frac{\sin \alpha}{1+\cos\alpha}$}
		\hspace{2mm}
		\fbox{$\displaystyle \textrm{ctg}\frac\alpha 2=\sqrt{\frac{1+\cos \alpha}{1-\cos\alpha}}=\frac{1+\cos \alpha}{\sin\alpha}=\frac{\sin \alpha}{1-\cos\alpha}$}
	\item \textbf{\textsf{Zbroj i razlika trigonometrijskih funkcija:}}\\
		\hspace*{5mm}\fbox{$\displaystyle \sin \alpha+\sin \beta=2\cdot \sin  \frac{\alpha +\beta}2 \cdot \cos\frac{\alpha-\beta}2$}
		\hspace{2mm}
		\fbox{$\displaystyle \sin \alpha-\sin \beta=2\cdot \cos \frac{\alpha+\beta}2 \cdot \sin\frac{\alpha-\beta}2$}\\
		\fbox{$\displaystyle \cos \alpha+\cos \beta=2\cdot \cos \frac{\alpha+\beta}2\cdot \cos\frac{\alpha-\beta}2$}
		\hspace{2mm}
		\fbox{$\displaystyle \cos \alpha-\cos \beta=-2\cdot \sin\frac{\alpha+\beta}2\cdot \sin\frac{\alpha-\beta}2$}\\
		\hspace*{10mm}\fbox{$\displaystyle \textrm{tg}\alpha\pm \textrm{tg}\beta=\frac{\sin(\alpha\pm\beta)}{\cos \alpha\cdot \cos \beta}$}
		\hspace{2mm}
		\fbox{$\displaystyle \textrm{ctg}\alpha\pm \textrm{tg}\beta=\pm\frac{\sin(\alpha\pm\beta)}{\sin \alpha\cdot \sin \beta}$}\\
		\hspace*{25mm}\fbox{$\displaystyle \textrm{tg}\alpha+ \textrm{ctg}\beta=\pm\frac{\cos(\alpha-\beta)}{\cos \alpha\cdot \sin \beta}$}
		\hspace{2mm}
		\fbox{$\displaystyle \textrm{ctg}\alpha- \textrm{tg}\beta=\pm\frac{\cos(\alpha \pm \beta)}{\sin \alpha\cdot \cos \beta}$}
	\item \textbf{\textsf{Produkt funkcija:}}\\
		\fbox{$\displaystyle \sin \alpha\sin \beta=\frac 12\left[ \cos(\alpha-\beta)-\cos(\alpha+\beta) \right]$}
		\hspace{2mm}
		\fbox{$\displaystyle \cos \alpha\cos \beta=\frac 12\left[ \cos(\alpha-\beta)+\cos(\alpha+\beta) \right]$}\\
		\hspace*{40mm}\fbox{$\displaystyle \sin \alpha\cos \beta=\frac 12\left[ \sin(\alpha-\beta)+\sin(\alpha+\beta) \right]$}
	\item \textbf{\textsf{\underline{Rje\v{s}avanje trokuta:}}}
		\item \textsf{Sinusov pou\v{c}ak:}\hspace{5mm}\fbox{$\displaystyle \frac a{\sin \alpha}=\frac b{\sin \beta}=\frac c{\sin \gamma}$} 
		\hspace{3mm} ili \hspace{3mm}
		\fbox{$\displaystyle a:b:c=\sin \alpha :\sin \beta :\sin\gamma$}
		\item \textsf{Kosinusov pou\v{c}ak:} \hspace{1mm}
		\fbox{$a^2=b^2+c^2-2bc\cdot \cos \alpha$} 
		\fbox{$b^2=a^2+c^2-2ac\cdot \cos \beta$} 
		\fbox{$c^2=a^2+b^2-2ab\cdot \cos \gamma$} \\
		\hspace*{33mm}
		\fbox{$\displaystyle \cos\alpha=\frac{b^2+c^2-a^2}{2bc}$}
		\hspace{7mm}
		\fbox{$\displaystyle \cos\beta=\frac{a^2+c^2-b^2}{2ac}$}
		\hspace{6mm}
		\fbox{$\displaystyle \cos\gamma=\frac{a^2+b^2-c^2}{2ab}$}
	\item \textbf{\textsf{\underline{Jednostavne trigonometrijske jednad\v{z}be:}}}\\
		\hspace*{34mm}
		\begin{tabular}{c|c}
		$\sin x=a$ &  $x=(-1)^k\sin^{-1}a+k\pi$\\
		\hline
		$\cos x=a$ &  $x=\pm\cos^{-1}a+2k\pi$\\
		\hline
		$\textrm{tg}\hspace{1mm} x=a$ &  $x=\textrm{tg}^{-1}a+k\pi$\\
		\hline
		$\textrm{ctg}\hspace{1mm} x=a$ &  $x=\textrm{ctg}^{-1}a+k\pi$\\
		\end{tabular}
\end{itemize}
