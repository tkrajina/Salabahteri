\fbox{{\large \textbf{Va\v{z}ne formule}}}
\begin{itemize}
\item umno\v{z}ak potencija s jednakim bazama: \hspace{1cm}\fbox{$a^n \cdot a^m=a^{n+m}$}
\item kvocjent potencija s jednakim bazama: \hspace{1cm}\fbox{$\frac{a^n}{a^m}=a^{n-m}$}
\item potencija produkta: \hspace{1cm} \fbox{$(a\cdot b)^n=a^n\cdot b^n$}
\item potencija kvocjenta: \hspace{1cm} \fbox{$\left(\frac a b \right)^n=\frac{a^n}{b^n}$}
\item potencija potencije: \hspace{1cm} \fbox{$\left( a^n \right)^m=a^{n\cdot m}$}
\item negativna potencija: \hspace{1cm} \fbox{$a^{-n}=\frac 1{a^n}$}
\item potencije s racionalnim eksponentom: \hspace{0.5cm}
\fbox{$a^{\frac 1 n}=\sqrt[n]{a}$} \hspace{1cm}
\fbox{$a^\frac mn =\sqrt[n]{a^m}=\left( \sqrt[n]{a} \right)^m$} \hspace{1cm}
\item razlika kvadrata: \hspace{1cm} \fbox{$a^2 - b^2 =(a+b)\cdot (a-b)$}
\item kvadrat binoma: \hspace{1cm} \fbox{$(a \pm b)^2 =a^2 \pm 2\cdot a\cdot b +b^2$}
\item rje\v{s}enje kvadratne jednad\v{z}be $a\cdot x^2+b\cdot x+c=0$: 
\hspace{1cm}\fbox{$\displaystyle x_{12}=\frac{-b\pm \sqrt{b^2-4ac}}{2a}$}
\end{itemize}
\fbox{{\large  \textbf{Binomni pou\v{c}ak}}}
\begin{itemize}
\item binomni koeficijent: \hspace{1cm} \fbox{$ {n \choose k} = \frac{n\cdot
(n-1)\cdot \dots \cdot (n-k+1)}{k!}$}
\item $n$-ta potencija binoma: \hspace{0.3cm}\fbox{$(a+b)^n={n \choose 0}a^n+{n\choose
1}a^{n-1}b^1+{n\choose 2}a^{n-1}b^2+\dots +{n\choose n-1}a^1b^{n-1}+{n\choose n}b^n$}
\item binomni koeficijenti se mogu izra\v{c}unati iz \emph{Pascalovog
trokuta}:\\ 
\hspace*{2cm}
\begin{tabular}{lcc}
1. & 1 1 & $1 \choose 0$ $1\choose 1$ \\
2. & 1 2 1 & $2 \choose 0$ $2 \choose 1$ $2 \choose 2$\\
3. & 1 3 3 1 & $3 \choose 0$ $3\choose 1$ $3\choose 2$ $3\choose 3$\\
4. & 1 4 6 4 1 & $4 \choose 0$ $4\choose 1$ $4\choose 2$ $4\choose 3$ $4 \choose 4$\\
5. & 1 5 10 10 5 1 & $5 \choose 0$ $5\choose 1$ $5\choose 2$ $5\choose 3$ $5 \choose 4$ $5 \choose 5$\\
6. & 1 6 15 20 15 6 1 & $6 \choose 0$ $6\choose 1$ $6\choose 2$ $6\choose 3$ $6 \choose 4$ $6 \choose 5$ $6 \choose 6$\\
\end{tabular}
\end{itemize}
\fbox{{\large  \textbf{Imaginarni brojevi}}}
\begin{itemize}
\item kompleksna jedinica: \hspace{1cm} \fbox{$i=\sqrt{-1}$}
\item op\'{c}i oblik kompleksnog broja: \hspace{1cm} \fbox{$z=a+i\cdot
b$}\\ \dots gdje je $a$ realni dio, a $b$ kompleksni dio broja $z$.
\item potencije broja $i$:
\hspace{0.5cm}\fbox{$i^{4n}=1$}
\hspace{1cm}
\fbox{$i^{4n+1}=i$}
\hspace{1cm}
\fbox{$i^{4n+2}=-1$}
\hspace{1cm}
\fbox{$i^{4n+3}=-i$}
\item trigonometrijski oblik kompleksnog broja: \hspace{1cm}\fbox{$z=r\cdot (sin\varphi +i\cdot cos \varphi )$}
\begin{itemize}
\item modul kompleksnog broja \hspace{1cm}\fbox{$r=\sqrt{a^2+b^2}$}
\item argument kompleksnog broja  \hspace{1cm}\fbox{$\textrm{tg}\varphi =\frac ba$}
\end{itemize} 
\item potenciranje kompleksnog broja u trigonometrijskom obliku: \hspace{1cm}\fbox{$\displaystyle z^n=\varphi ^n\cdot (\cos n\varphi +i\cdot \sin n\varphi)$}
\item $n$-ti korijeni kompleksnog broja $z$: 
\hspace{0.3cm}\fbox{$\displaystyle \sqrt[n]{z}=\sqrt[n]{\varphi}\cdot (\cos \frac{\varphi +2k \pi}n + i\cdot \sin \frac{\varphi +2k \pi}n),\hspace{0.3cm}k=1,2,\dots n-1$}
\end{itemize}

\fbox{{\large  \textbf{Nizovi}}}\\
\begin{itemize}
\item $a_n$ je vrijednost $n$-tog \v{c}lana niza,
\item $d$ -- korak aritmeti\v{c}kog niza
\item $q$ -- korak geometrijskog niza
\item $S_n$ zbroj prvih $n$ \v{c}lanova niza
\end{itemize}
\hspace*{2cm}
\begin{tabular}{l|c|c}
& {\large aritmeti\v{c}ki niz} & {\large geometrijski niz}\\
\hline
op\'{c}i \v{c}lan: & $a_n=a_1+d\cdot (n-1)$ & $a_n=a_1\cdot q^{n-1}$\\
\hline
va\v{z}no svojstvo: & $a_n-a_{n-1}=d$ & $\frac{a_n}{a_{n-1}}=q$ \\
\hline
suma prvih $n$ \v{c}lanova: & $S_n=\frac{n\cdot (a_1+a_n)}2 $ &
$S_n=\frac{a_1\cdot (q^n-1)}{q-1}$ \\
\end{tabular}

\fbox{{\large  \textbf{Limes niza}}}
\begin{itemize}
\item limes sume: \hspace{1cm} \fbox{$\displaystyle \lim_{n \to \infty}(a_n\pm b_n)=\lim_{n \to \infty}a_n\pm \lim_{n \to \infty}b_n$}
\item limes produkta: \hspace{1cm} \fbox{$\displaystyle \lim_{n \to
\infty}(a_n\cdot  b_n)=\lim_{n \to \infty}a_n\cdot  \lim_{n \to \infty}b_n$}
\item limes kvocijenta: \hspace{1cm} \fbox{$\displaystyle \lim_{n \to
\infty}\frac{a_n}{b_n}=\frac{\lim_{n \to \infty}a_n}{\lim_{n \to \infty}b_n}$}
\item \fbox{$\displaystyle \lim_{n \to \infty}(c\cdot a_n)=c\cdot \lim_{n \to \infty}a_n$}
\item \fbox{$\displaystyle \lim_{n \to \infty}\frac 1 n=0$}
\hspace{1cm}\fbox{$\displaystyle \lim_{n \to \infty}\frac 1 {n^k}=0$}
\item \fbox{$\displaystyle \lim_{n \to \infty}q^n=\left\{
	\begin{array}{ll}
	\infty & \textrm{ako je $q>1$}\\
	1 & \textrm{ako je $q=1$}\\
	0 & \textrm{ako je $-1<q<1$}\\
	\textrm{divergira} & \textrm{ako je $q\leq -1$}\\
	\end{array} \right.$}
\end{itemize}

\fbox{{\large  \textbf{Derivacije}}}
\begin{itemize}
\item definicija derivacije funkcije u to\v{c}ki $x$:\hspace{1cm}
\fbox{$\displaystyle f'(x)=\lim_{\Delta x \to 0}\frac{f(x+\Delta x)-f(x)}{\Delta x}$}
\item derivacija funkcije u to\v{c}ki je jednaka koeficijentu smjera tangente koja prolazi tom to\v{c}kom na grafu te funkcije
\item derivacija sume/razlike: \hspace{1cm}
\fbox{$(f\pm g)'=f'\pm g'$}
\item derivacija produkta i kvocjenta:\\
\hspace*{1cm} 
	\fbox{$(f\cdot g)'=f'\cdot g+f\cdot g'$}
\hspace{1cm}
	\fbox{$(c\cdot f)'=c\cdot f'$}
\hspace{1cm}
	\fbox{$\displaystyle \left(\frac fg \right)'=\frac{f'\cdot g-f\cdot g'}{g^2}$}
\item derivacija kompozicije funkcija: \hspace{1cm}\fbox{$(f\circ g)'=(f'\circ g)\cdot g'$}
\item derivacija konstante, potencije i korijena: 
\fbox{$c'=0$}
\hspace{1cm} \fbox{$(x^n)'=n\cdot x^{n-1}$}
\hspace{1cm} \fbox{$(\sqrt x)'=\frac 1{2\sqrt{x}}$}
\item derivacija trigonometrijskih funkcija:\\
\hspace*{1.5cm}\fbox{$(\sin x)'=\cos x$}
\hspace{0.5cm}\fbox{$(\cos x)' =-\sin x$}
\hspace{0.5cm}\fbox{$(\textrm{tg}x)'=\frac 1{\cos^2 x}$}
\hspace{0.5cm}\fbox{$(\textrm{ctg}x)'=-\frac 1{\sin^2 x}$}
\item derivacija eksponencijalne i logaritamske funkcije:\\
\hspace*{1cm} \fbox{$(e^x)'=e^x$}
\hspace{2cm} \fbox{$(a^x)'=a^x\cdot \ln a$}\\
\hspace*{2cm} \fbox{$\displaystyle (\ln x)'=\frac 1x$}
\hspace{2cm} \fbox{$\displaystyle (\log_a x)'=\frac 1{x\cdot \ln a}$}
\end{itemize}
